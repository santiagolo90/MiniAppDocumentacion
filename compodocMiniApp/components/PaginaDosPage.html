  irTutorial(donde:string){
    if (donde == "alta") {
      let profileModal = this.modalCtrl.create(TutoPage, { tipo: "alta"});
      profileModal.present();
    }
    if (donde == "baja") {
      let profileModal = this.modalCtrl.create(TutoPage, { tipo: "baja"});
      profileModal.present();
    }
    if (donde == "modificacion") {
      let profileModal = this.modalCtrl.create(TutoPage, { tipo: "modificacion"});
      profileModal.present();
    }
  }
